def run(command)
  exitcode = nil
  Open3.popen3(command) do |stdin, stdout, stderr, thread|
    pid = thread.pid
    debug1(stdout.read.chomp)
    debug1(stderr.read.chomp)
    exitcode = thread.value
  end
  exitcode
end
